Loading data from files

CSV or Tab delimited files

Remote JSON files

CanvasXpress PNG files

CSV file

      
var cX = new CanvasXpress(
  {
   renderTo: "canvasId",
   data: "https://raw.githubusercontent.com/neuhausi/Rdatasets/master/csv/datasets/ToothGrowth.csv",
   config: {
     title: "Tooth Growth",
     smpLabelRotate: 90,
     graphType: "Boxplot",
     graphOrientation: "vertical",
     transposeData: true,
     asVariableFactors: [ "dose" ],
     stringSampleFactors: [ "dose" ],
     groupingFactors: [ "dose" ]
   },
   events: false
  }
);
