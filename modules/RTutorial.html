CanvasXpress in R Console

      




  library(canvasXpress)
  data = t(iris[,1:4])
  varAnnot = as.matrix(iris[,5])
  colnames(varAnnot) = "Species"
  canvasXpress(
    data = t(data),
    varAnnot = varAnnot,
    colorBy = "Species",
    graphType = "Scatter2D",
    scatterPlotMatrix = TRUE,
    theme = "CanvasXpress"
  )
