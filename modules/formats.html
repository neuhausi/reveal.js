Supported Data Formats

  • Native - JSON object
    • X,Y,Z graphs
    • Networks
    • Genome Browser
    • Venn Diagrams
  • Delimited text files
  • XML files
  • CanvasXpress - PNG files

Native JSON Format

      
var data = {
  "y": {
    "vars": [ "Gene1"],
    "smps": [ "Smp1", "Smp2", "Smp3" ],
    "data": [ [ 10, 35, 88 ] ]
  }
};

var conf = {
  "graphType": "Bar"
}

var cX = new CanvasXpress("canvasId", data, conf);

Parameters: (1) DOM id, (2) data, (3) config

Alternate JSON Format

      
var cX = new CanvasXpress("canvasId", {
    "y": {
      "vars": [ "Variable1", "Variable2" ],
      "smps": [ "Sample1", "Sample2", "Sample3" ],
      "data": [ [ 10, 20, 30 ],
              [   35, 25, 15 ] ]
    },
    "x": {
      "Tissue": [ "Kidney", "Lung", "Heart" ],
      "Donor":  [ "D1",     "D1",   "D2" ]
    },
    "z": {
      "Symbol":  [ "AAA", "BBB" ],
      "Pathway": [ "P1",  "P2" ]
    }
  }, {
    "graphType": "Bar"
  }
);

Parameters: (1) DOM id, (2) data, (3) config
