Usage in the Web

      
<html>

  <head>  
    <link rel="stylesheet" href="path-to-canvasXpress.css" type="text/css"/>
    <script type="text/javascript" src="path-to-canvasXpress.min.js"></script>
    <script>
      onReady(function () { 
        var cX = new CanvasXpress({
          "renderTo": "canvasId", 
          "data": {
            "y": {
              "vars": [ "Gene1"],
              "smps": [ "Smp1", "Smp2", "Smp3" ],
              "data": [ [ 10, 35, 88 ] ]
            }
          },
          "config": {
            "graphType": "Bar"
          }
        });
      });
    </script>
  </head>
  
  <body>
    <canvas id="canvasId" width="540" height="540"></canvas>    
  </body>
  
</html>

Data is 'L' shaped

Each container is a data frame (like in R)

One Series Data Set

                
{
  "y": {
    "vars": [ "Variable1" ],
    "smps": [ "Sample1", "Sample2", "Sample3" ],
    "data": [ [ 10, 20, 30 ] ]
  },
  "x": {
    "Tissue": [ "Kidney", "Lung", "Heart" ],
    "Donor":  [ "D1",     "D1",   "D2" ]
  },
  "z": {
    "Symbol":  [ "AAA" ],
    "Pathway": [ "P1" ]
  }
}

Two Series Data Set

                
{
  "y": {
    "vars": [ "Variable1", "Variable2" ],
    "smps": [ "Sample1", "Sample2", "Sample3" ],
    "data": [ [ 10, 20, 30 ],
            [   35, 25, 15 ] ]
  },
  "x": {
    "Tissue": [ "Kidney", "Lung", "Heart" ],
    "Donor":  [ "D1",     "D1",   "D2" ]
  },
  "z": {
    "Symbol":  [ "AAA", "BBB" ],
    "Pathway": [ "P1",  "P2" ]
  }
}

Two Series Example

      
<html>

  <head>  
    <link rel="stylesheet" href="path-to-canvasXpress.css" type="text/css"/>
    <script type="text/javascript" src="path-to-canvasXpress.min.js"></script>
    <script>
      onReady(function () { 
        var cX = new CanvasXpress({
          "renderTo": "canvasId", 
          "data": {
            "y": {
              "vars": [ "Variable1", "Variable2"],
              "smps": [ "Sample1", "Sample2", "Sample3" ],
              "data": [ [ 10, 20, 30 ],
                        [ 35, 25, 15 ] ]
            },
            "x": {
              "Tissue" : ["Kidney", "Lung", "Heart"]
            }
          },
          "config": {
            "graphType": "Bar",
            "smpOverlays": ["Tissue"]
          }
        });
      });
    </script>
  </head>
  
  <body>
    <canvas id="canvasId" width="540" height="540"></canvas>    
  </body>
  
</html>
